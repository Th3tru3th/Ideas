     System.out.println("Welcome to my Budgeting Program");

     Scanner input = new Scanner(System.in);
     double budgetAmt = input.nextDouble();
     System.out.println("You have entered: " + budgetAmt);
     System.out.println("What percentage would you like taken out? ");
     double percentage = input.nextDouble();
     
     percentage = percentage / 100;

     double percent1 = budgetAmt * percentage;
     System.out.println("That equals " + percent1);

    // calculate the remaining amount.
    
    double remain = budgetAmt - percent1;
